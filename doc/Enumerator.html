Properties

obj

Receiver object of Enumerator. Can get this.obj any time. But should not be change this.

method

Caller of Enumerator. to run this function for each cycle.

index

Use {`next`,`next_values`,`peek`,`peek_values`} function index.

Instance Methods

initialize (obj, method) -> Enumerator

Enumerator initializer

[PARAM] obj:
Enumerator receiver object
[PARAM] method:
execute method when execute `each`. if this is nothing set default method by `obj`
  var e = Enumerator([1,1,2,2,3]);
  e.each(function(i){
    console.log(i); //=> 1,1,2,2,3
  });
  Enumerator(function(y){
      y('first');
      y('second');
      y('third');
  }).each(function(i){
      console.log(i + ' next'); //=> 'first next', 'second next', 'third next'
  });
  var e = Enumerator([1,2,3], Array.prototype.map);
  var array = e.each(function(i){
    return i * i;
  }));
  // array => [1,4,9]
  each() -> this
  each(callback(item)) -> *

Start run enumerator function.

[PARAM] callback: 
  Enumerator([1,2,3]).each(function(i){
    console.log(i); //=> 1,2,3
  });
next () -> *

Return object from Receiver object at now `index` and `index` increment. 

  var e = Enumerator([1,2,3]);
  e.next(); //=> 1
  e.next(); //=> 2
  e.next(); //=> 3
  e.next(); //=> throw StopIteration
next_values () -> Array

Return Array that object by `next` put in. this function judgment to use call yield arguments nothing or undefined. 

  var e = Enumerator(function(y){
    y();
    y(undefined);
    y(null);
  });
  e.next_values(); //=> []
  e.next_values(); //=> [undefined]
  e.next_values(); //=> [null]
peek () -> *

This is same `next`. but not increment `index`. 

  var e = Enumerator([1,2,3]);
  e.peek(); //=> 1
  e.peek(); //=> 1
  e.peek(); //=> 1
peek_values () -> Array

This is same `next_values`. but not increment `index`. 

  var e = Enumerator(function(y){
    y();
    y(undefined);
    y(null);
  });
  e.peek_values(); //=> []
  e.peek_values(); //=> []
  e.peek_values(); //=> []
rewind () -> Enumerator

Reset `index` if `obj` have rewind function then call it.

with_index (offset, callback) -> Enumerator|*

Loop with index number. index start `offset`(default 0).

[PARAM] offset:
[PARAM] callback: 
  var e = Enumerator([1,2,3]).with_index();
  e.each(function(i, index){
    console.log([i, index]); //=> [1,0], [2,1], [3,2]
  });
  Enumerator([1,2,3]).with_index(10,function(i, index){
    console.log([i, index]); //=> [1,10], [2,11], [3,12]
  });
  with_object(object) -> object
  each_with_object(object) -> object

Loop with any object. `obj` is cloned and keep value in `with_object` function.

[PARAM] obj:
[PARAM] callback: 
  var a = [];
  var e = Enumerator([1,2,3]).with_object(a);
  e.each(function(i, obj){
    obj.push(i);
    console.log([i, obj]); //=> [1,[1]], [2,[1,2]], [3,[1,2,3]]
  });
  console.log(a); //=> []
  Enumerator([1,2,3]).with_object(a,function(i, obj){
    obj.push(i);
    console.log([i, obj]); //=> [1,[1]], [2,[1,2]], [3,[1,2,3]]
  });
  console.log(a); //=> []

parallel (enumerators) -> Enumerator

Parallel run enumerator loops

[PARAM] enumerators: 
  Enumerator.parallel([
    [1,2,3],             // something object
    {'a':1,'b':2},       // can use Array or Object
    Enumerator([4,5,6]), // or Enumerator object
  ]).each(function(values){
    console.log(values); //=> [1,['a',1],4], [2,['b',2],5], [3,undefined,6]
  });
